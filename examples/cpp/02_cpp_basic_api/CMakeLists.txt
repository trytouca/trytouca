include(GNUInstallDirs)

add_executable(example_cpp_basic_api "")

target_sources(
        example_cpp_basic_api
    PRIVATE
        students.cpp
        students.hpp
        students_test.cpp
        students_test.hpp
)

target_link_libraries(
        example_cpp_basic_api
    PRIVATE
        touca_client
)

    TREE ${CMAKE_CURRENT_LIST_DIR}
    FILES $<TARGET_PROPERTY:example_cpp_basic_api,SOURCES>
)

install(
    TARGETS example_cpp_basic_api
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
)
